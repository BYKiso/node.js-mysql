262  eaw.characterLength = function(character) {
263    var code = this.eastAsianWidth(character);
264    if (code == 'F' || code == 'W' || code == 'A') {
265      return 2;
266    } else {
267      return 1;
268    }
269  };

271  // Split a string considering surrogate-pairs.
272  function stringToArray(string) {
273    return string.match(/[\uD800-\uDBFF][\uDC00-\uDFFF]|[^\uD800-\uDFFF]/g) || [];
274  }

276  eaw.length = function(string) {
277    var characters = stringToArray(string);
278    var len = 0;
279    for (var i = 0; i < characters.length; i++) {
280      len = len + this.characterLength(characters[i]);
281    }

285  eaw.slice = function(text, start, end) {
286    textLen = eaw.length(text)
287    start = start ? start : 0;
288    end = end ? end : 1;
289    if (start < 0) {
290        start = textLen + start;
291    }
292    if (end < 0) {
293        end = textLen + end;
294    }
295    var result = '';
296    var eawLen = 0;
297    var chars = stringToArray(text);
298    for (var i = 0; i < chars.length; i++) {
299      var char = chars[i];
300      var charLen = eaw.length(char);
301      if (eawLen >= start - (charLen == 2 ? 1 : 0)) {
302          if (eawLen + charLen <= end) {
303              result += char;
304          } else {

306          }
307      }
308      eawLen += charLen;
309    }
