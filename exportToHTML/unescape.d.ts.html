1    import { MinimatchOptions } from './index.js';
2    /**
3     * Un-escape a string that has been escaped with {@link escape}.
4     *
5     * If the {@link windowsPathsNoEscape} option is used, then square-brace
6     * escapes are removed, but not backslash escapes.  For example, it will turn
7     * the string `'[*]'` into `*`, but it will not turn `'\\*'` into `'*'`,
8     * becuase `\` is a path separator in `windowsPathsNoEscape` mode.
9     *
10    * When `windowsPathsNoEscape` is not set, then both brace escapes and
11    * backslash escapes are removed.
12    *
13    * Slashes (and backslashes in `windowsPathsNoEscape` mode) cannot be escaped
14    * or unescaped.
15    */
16   export declare const unescape: (s: string, { windowsPathsNoEscape, }?: Pick<MinimatchOptions, "windowsPathsNoEscape">) => string;
