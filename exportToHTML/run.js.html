22   function run(options) {
23     var cmd = config.command.raw;
24     // moved up
25     // we need restart function below in the global scope for run.kill
26     /*jshint validthis:true*/
27     restart = run.bind(this, options);
28     run.restart = restart;
29   
30     // binding options with instance of run
31     // so that we can use it in run.kill
32     run.options = options;
33   
34     var runCmd = !options.runOnChangeOnly || config.lastStarted !== 0;
35     if (runCmd) {
36       utils.log.status('starting `' + config.command.string + '`');
37     } else {
38       // should just watch file if command is not to be run
39       // had another alternate approach
40       // to stop process being forked/spawned in the below code
41       // but this approach does early exit and makes code cleaner
42       debug('start watch on: %s', config.options.watch);
43       if (config.options.watch !== false) {
44         watch();
45         return;
46       }
47     }
48   
49     config.lastStarted = Date.now();
50   
51     var stdio = ['pipe', 'pipe', 'pipe'];
52   
53     if (config.options.stdout) {
54       stdio = ['pipe', process.stdout, process.stderr];
55     }
56   
57     if (config.options.stdin === false) {
58       stdio = [process.stdin, process.stdout, process.stderr];
59     }
60   
61     var sh = 'sh';
62     var shFlag = '-c';
63   
64     const binPath = process.cwd() + '/node_modules/.bin';
65   
66     const spawnOptions = {
67       env: Object.assign({}, options.execOptions.env, process.env, {
68         PATH:
69           binPath +
70           path.delimiter +
71           (undefsafe(options, '.execOptions.env.PATH') || process.env.PATH),
72       }),
73       stdio: stdio,
74     };
75   
76     var executable = cmd.executable;
77   
78     if (utils.isWindows) {
79       // if the exec includes a forward slash, reverse it for windows compat
80       // but *only* apply to the first command, and none of the arguments.
81       // ref #1251 and #1236
82       if (executable.indexOf('/') !== -1) {
83         executable = executable
84           .split(' ')
85           .map((e, i) => {
86             if (i === 0) {
87               return path.normalize(e);
88             }
89             return e;
90           })
91           .join(' ');
92       }
93       // taken from npm's cli: https://git.io/vNFD4
94       sh = process.env.comspec || 'cmd';
95       shFlag = '/d /s /c';
96       spawnOptions.windowsVerbatimArguments = true;
97       spawnOptions.windowsHide = true;
98     }
99   
100    var args = runCmd ? utils.stringify(executable, cmd.args) : ':';
101    var spawnArgs = [sh, [shFlag, args], spawnOptions];
102  
103    const firstArg = cmd.args[0] || '';
104  
105    var inBinPath = false;
106    try {
107      inBinPath = statSync(`${binPath}/${executable}`).isFile();
108    } catch (e) {}
109  
110    // hasStdio allows us to correctly handle stdin piping
111    // see: https://git.io/vNtX3
112    const hasStdio = utils.satisfies('>= 6.4.0 || < 5');
113  
114    // forking helps with sub-process handling and tends to clean up better
115    // than spawning, but it should only be used under specific conditions
116    const shouldFork =
117      !config.options.spawn &&
118      !inBinPath &&
119      !(firstArg.indexOf('-') === 0) && // don't fork if there's a node exec arg
120      firstArg !== 'inspect' && // don't fork it's `inspect` debugger
121      executable === 'node' && // only fork if node
122      utils.version.major > 4; // only fork if node version > 4
123  
124    if (shouldFork) {
125      // this assumes the first argument is the script and slices it out, since
126      // we're forking
127      var forkArgs = cmd.args.slice(1);
128      var env = utils.merge(options.execOptions.env, process.env);
129      stdio.push('ipc');
130      const forkOptions = {
131        env: env,
132        stdio: stdio,
133        silent: !hasStdio,
134      };
135      if (utils.isWindows) {
136        forkOptions.windowsHide = true;
137      }
138      child = fork(options.execOptions.script, forkArgs, forkOptions);
139      utils.log.detail('forking');
140      debug('fork', sh, shFlag, args);
141    } else {
142      utils.log.detail('spawning');
143      child = spawn.apply(null, spawnArgs);
144      debug('spawn', sh, shFlag, args);
145    }
146  
147    if (config.required) {
148      var emit = {
149        stdout: function (data) {
150          bus.emit('stdout', data);
151        },
152        stderr: function (data) {
153          bus.emit('stderr', data);
154        },
155      };
156  
157      // now work out what to bind to...
158      if (config.options.stdout) {
159        child.on('stdout', emit.stdout).on('stderr', emit.stderr);
160      } else {
161        child.stdout.on('data', emit.stdout);
162        child.stderr.on('data', emit.stderr);
163  
164        bus.stdout = child.stdout;
165        bus.stderr = child.stderr;
166      }
167  
168      if (shouldFork) {
169        child.on('message', function (message, sendHandle) {
170          bus.emit('message', message, sendHandle);
171        });
172      }
173    }
174  
175    bus.emit('start');
176  
177    utils.log.detail('child pid: ' + child.pid);
178  
179    child.on('error', function (error) {
180      bus.emit('error', error);
181      if (error.code === 'ENOENT') {
182        utils.log.error('unable to run executable: "' + cmd.executable + '"');
183        process.exit(1);
184      } else {
185        utils.log.error('failed to start child process: ' + error.code);
186        throw error;
187      }
188    });
189  
190    child.on('exit', function (code, signal) {
191      if (child && child.stdin) {
192        process.stdin.unpipe(child.stdin);
193      }
194  
195      if (code === 127) {
196        utils.log.error(
197          'failed to start process, "' + cmd.executable + '" exec not found'
198        );
199        bus.emit('error', code);
200        process.exit();
201      }
202  
203      // If the command failed with code 2, it may or may not be a syntax error
204      // See: http://git.io/fNOAR
205      // We will only assume a parse error, if the child failed quickly
206      if (code === 2 && Date.now() < config.lastStarted + 500) {
207        utils.log.error('process failed, unhandled exit code (2)');
208        utils.log.error('');
209        utils.log.error('Either the command has a syntax error,');
210        utils.log.error('or it is exiting with reserved code 2.');
211        utils.log.error('');
212        utils.log.error('To keep nodemon running even after a code 2,');
213        utils.log.error('add this to the end of your command: || exit 1');
214        utils.log.error('');
215        utils.log.error('Read more here: https://git.io/fNOAG');
216        utils.log.error('');
217        utils.log.error('nodemon will stop now so that you can fix the command.');
218        utils.log.error('');
219        bus.emit('error', code);
220        process.exit();
221      }
222  
223      // In case we killed the app ourselves, set the signal thusly
224      if (killedAfterChange) {
225        killedAfterChange = false;
226        signal = config.signal;
227      }
228      // this is nasty, but it gives it windows support
229      if (utils.isWindows && signal === 'SIGTERM') {
230        signal = config.signal;
231      }
232  
233      if (signal === config.signal || code === 0) {
234        // this was a clean exit, so emit exit, rather than crash
235        debug('bus.emit(exit) via ' + config.signal);
236        bus.emit('exit', signal);
237  
238        // exit the monitor, but do it gracefully
239        if (signal === config.signal) {
240          return restart();
241        }
242  
243        if (code === 0) {
244          // clean exit - wait until file change to restart
245          if (runCmd) {
246            utils.log.status('clean exit - waiting for changes before restart');
247          }
248          child = null;
249        }
250      } else {
251        bus.emit('crash');
252        if (options.exitcrash) {
253          utils.log.fail('app crashed');
254          if (!config.required) {
255            process.exit(1);
256          }
257        } else {
258          utils.log.fail(
259            'app crashed - waiting for file changes before' + ' starting...'
260          );
261          child = null;
262        }
263      }
264  
265      if (config.options.restartable) {
266        // stdin needs to kick in again to be able to listen to the
267        // restart command
268        process.stdin.resume();
269      }
270    });
271  
272    // moved the run.kill outside to handle both the cases
273    // intial start
274    // no start
275  
276    // connect stdin to the child process (options.stdin is on by default)
277    if (options.stdin) {
278      process.stdin.resume();
279      // FIXME decide whether or not we need to decide the encoding
280      // process.stdin.setEncoding('utf8');
281  
282      // swallow the stdin error if it happens
283      // ref: https://github.com/remy/nodemon/issues/1195
284      if (hasStdio) {
285        child.stdin.on('error', () => {});
286        process.stdin.pipe(child.stdin);
287      } else {
288        if (child.stdout) {
289          child.stdout.pipe(process.stdout);
290        } else {
291          utils.log.error(
292            'running an unsupported version of node ' + process.version
293          );
294          utils.log.error(
295            'nodemon may not work as expected - ' +
296              'please consider upgrading to LTS'
297          );
298        }
299      }
300  
301      bus.once('exit', function () {
302        if (child && process.stdin.unpipe) {
303          // node > 0.8
304          process.stdin.unpipe(child.stdin);
305        }
306      });
307    }
308  
309    debug('start watch on: %s', config.options.watch);
310    if (config.options.watch !== false) {
311      watch();
312    }
313  }

315  function waitForSubProcesses(pid, callback) {
316    debug('checking ps tree for pids of ' + pid);
317    psTree(pid, (err, pids) => {
318      if (!pids.length) {
319        return callback();
320      }
321  
322      utils.log.status(
323        `still waiting for ${pids.length} sub-process${
324          pids.length > 2 ? 'es' : ''
325        } to finish...`
326      );
327      setTimeout(() => waitForSubProcesses(pid, callback), 1000);
328    });
329  }

331  function kill(child, signal, callback) {
332    if (!callback) {
333      callback = noop;
334    }
335  
336    if (utils.isWindows) {
337      const taskKill = () => {
338        try {
339          exec('taskkill /pid ' + child.pid + ' /T /F');
340        } catch (e) {
341          utils.log.error('Could not shutdown sub process cleanly');
342        }
343      };
344  
345      // We are handling a 'SIGKILL' , 'SIGUSR2' and 'SIGUSR1' POSIX signal under Windows the
346      // same way it is handled on a UNIX system: We are performing
347      // a hard shutdown without waiting for the process to clean-up.
348      if (
349        signal === 'SIGKILL' ||
350        osRelease < 10 ||
351        signal === 'SIGUSR2' ||
352        signal === 'SIGUSR1'
353      ) {
354        debug('terminating process group by force: %s', child.pid);
355  
356        // We are using the taskkill utility to terminate the whole
357        // process group ('/t') of the child ('/pid') by force ('/f').
358        // We need to end all sub processes, because the 'child'
359        // process in this context is actually a cmd.exe wrapper.
360        taskKill();
361        callback();
362        return;
363      }
364  
365      try {
366        // We are using the Windows Management Instrumentation Command-line
367        // (wmic.exe) to resolve the sub-child process identifier, because the
368        // 'child' process in this context is actually a cmd.exe wrapper.
369        // We want to send the termination signal directly to the node process.
370        // The '2> nul' silences the no process found error message.
371        const resultBuffer = execSync(
372          `wmic process where (ParentProcessId=${child.pid}) get ProcessId 2> nul`
373        );
374        const result = resultBuffer.toString().match(/^[0-9]+/m);
375  
376        // If there is no sub-child process we fall back to the child process.
377        const processId = Array.isArray(result) ? result[0] : child.pid;
378  
379        debug('sending kill signal SIGINT to process: %s', processId);
380  
381        // We are using the standalone 'windows-kill' executable to send the
382        // standard POSIX signal 'SIGINT' to the node process. This fixes #1720.
383        const windowsKill = path.normalize(
384          `${__dirname}/../../bin/windows-kill.exe`
385        );
386  
387        // We have to detach the 'windows-kill' execution completely from this
388        // process group to avoid terminating the nodemon process itself.
389        // See: https://github.com/alirdn/windows-kill#how-it-works--limitations
390        //
391        // Therefore we are using 'start' to create a new cmd.exe context.
392        // The '/min' option hides the new terminal window and the '/wait'
393        // option lets the process wait for the command to finish.
394  
395        execSync(
396          `start "windows-kill" /min /wait "${windowsKill}" -SIGINT ${processId}`
397        );
398      } catch (e) {
399        taskKill();
400      }
401      callback();
402    } else {
403      // we use psTree to kill the full subtree of nodemon, because when
404      // spawning processes like `coffee` under the `--debug` flag, it'll spawn
405      // it's own child, and that can't be killed by nodemon, so psTree gives us
406      // an array of PIDs that have spawned under nodemon, and we send each the
407      // configured signal (default: SIGUSR2) signal, which fixes #335
408      // note that psTree also works if `ps` is missing by looking in /proc
409      let sig = signal.replace('SIG', '');
410  
411      psTree(child.pid, function (err, pids) {
412        // if ps isn't native to the OS, then we need to send the numeric value
413        // for the signal during the kill, `signals` is a lookup table for that.
414        if (!psTree.hasPS) {
415          sig = signals[signal];
416        }
417  
418        // the sub processes need to be killed from smallest to largest
419        debug('sending kill signal to ' + pids.join(', '));
420  
421        child.kill(signal);
422  
423        pids.sort().forEach((pid) => exec(`kill -${sig} ${pid}`, noop));
424  
425        waitForSubProcesses(child.pid, () => {
426          // finally kill the main user process
427          exec(`kill -${sig} ${child.pid}`, callback);
428        });
429      });
430    }
431  }

433  run.kill = function (noRestart, callback) {
434    // I hate code like this :(  - Remy (author of said code)
435    if (typeof noRestart === 'function') {
436      callback = noRestart;
437      noRestart = false;
438    }
439  
440    if (!callback) {
441      callback = noop;
442    }
443  
444    if (child !== null) {
445      // if the stdin piping is on, we need to unpipe, but also close stdin on
446      // the child, otherwise linux can throw EPIPE or ECONNRESET errors.
447      if (run.options.stdin) {
448        process.stdin.unpipe(child.stdin);
449      }
450  
451      // For the on('exit', ...) handler above the following looks like a
452      // crash, so we set the killedAfterChange flag if a restart is planned
453      if (!noRestart) {
454        killedAfterChange = true;
455      }
456  
457      /* Now kill the entire subtree of processes belonging to nodemon */
458      var oldPid = child.pid;
459      if (child) {
460        kill(child, config.signal, function () {
461          // this seems to fix the 0.11.x issue with the "rs" restart command,
462          // though I'm unsure why. it seems like more data is streamed in to
463          // stdin after we close.
464          if (child && run.options.stdin && child.stdin && oldPid === child.pid) {
465            child.stdin.end();
466          }
467          callback();
468        });
469      }
470    } else if (!noRestart) {
471      // if there's no child, then we need to manually start the process
472      // this is because as there was no child, the child.on('exit') event
473      // handler doesn't exist which would normally trigger the restart.
474      bus.once('start', callback);
475      run.restart();
476    } else {
477      callback();
478    }
479  };

481  run.restart = noop;

483  bus.on('quit', function onQuit(code) {
484    if (code === undefined) {
485      code = 0;
486    }
487  
488    // remove event listener
489    var exitTimer = null;
490    var exit = function () {
491      clearTimeout(exitTimer);
492      exit = noop; // null out in case of race condition
493      child = null;
494      if (!config.required) {
495        // Execute all other quit listeners.
496        bus.listeners('quit').forEach(function (listener) {
497          if (listener !== onQuit) {
498            listener();
499          }
500        });
501        process.exit(code);
502      } else {
503        bus.emit('exit');
504      }
505    };
506  
507    // if we're not running already, don't bother with trying to kill
508    if (config.run === false) {
509      return exit();
510    }
511  
512    // immediately try to stop any polling
513    config.run = false;
514  
515    if (child) {
516      // give up waiting for the kids after 10 seconds
517      exitTimer = setTimeout(exit, 10 * 1000);
518      child.removeAllListeners('exit');
519      child.once('exit', exit);
520  
521      kill(child, 'SIGINT');
522    } else {
523      exit();
524    }
525  });

527  bus.on('restart', function () {
528    // run.kill will send a SIGINT to the child process, which will cause it
529    // to terminate, which in turn uses the 'exit' event handler to restart
530    run.kill();
531  });

533  // remove the child file on exit
534  process.on('exit', function () {
535    utils.log.detail('exiting');
536    if (child) {
537      child.kill();
538    }
